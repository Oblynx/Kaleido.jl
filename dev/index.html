Setting/getting multiple locations

Kaleido.MultiLens — Type.
MultiLens(lenses::Tuple)
MultiLens(lenses::NamedTuple)

Examples

julia> using Setfield, Kaleido

julia> ml = MultiLens((
           (@lens _.x),
           (@lens _.y.z),
       ));

julia> @assert get((x=1, y=(z=2,)), ml) === (1, 2)

julia> @assert set((x=1, y=(z=2,)), ml, ("x", "y.z")) === (x="x", y=(z="y.z",))

julia> ml = MultiLens((
           a = (@lens _.x),
           b = (@lens _.y.z),
       ));

julia> @assert get((x=1, y=(z=2,)), ml) === (a=1, b=2)

julia> @assert set((x=1, y=(z=2,)), ml, (a=:x, b="y.z")) === (x=:x, y=(z="y.z",))

julia> @assert set((x=1, y=(z=2,)), ml, (b="y.z", a=:x)) === (x=:x, y=(z="y.z",))

Bijective transformation as lenses

Kaleido.BijectionLens — Type.
BijectionLens(inward, outward) :: Lens
BijectionLens(xf::TransformVariables.AbstractTransform) :: Lens

Examples

julia> using Setfield, Kaleido

julia> l = (@lens _.y[2]) ∘ BijectionLens(x -> 2x, x -> x/2);

julia> obj = (x=0, y=(1, 2, 3));

julia> @assert get(obj, l) == 1.0 == 2/2

julia> @assert set(obj, l, 0.5) == (x=0, y=(1, 1.0, 3))
julia> using Setfield, Kaleido, TransformVariables

julia> l = (@lens _.y[2]) ∘ BijectionLens(as𝕀);

julia> obj = (x=0, y=(1, 0.5, 3));

julia> @assert get(obj, l) == 0

julia> @assert set(obj, l, Inf).y[2] ≈ 1

julia> @assert set(obj, l, -Inf).y[2] ≈ 0.0
